CREATE PROCEDURE [dbo].[BOOK_ISSUE_TO_STUDENT]
	@BID  AS INT
AS
BEGIN
	
	SET NOCOUNT ON;

	UPDATE BookMst SET AvailableQnt=AvailableQnt-1, RentQnt=RentQnt+1 WHERE BookID=@BID
END
